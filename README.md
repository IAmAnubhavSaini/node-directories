# node-directories

Get directories - node module

## Usage

```javascript

var d = require('node-directories')

d()
    ['Applications',
    'Developer',
    'Library',
    'Network',
    'System',
    'Users',
    'Volumes',
    'bin',
    'cores',
    'dev',
    'home',
    'net',
    'opt',
    'private',
    'sbin',
    'usr']

d('./')
    ['.git']

```
